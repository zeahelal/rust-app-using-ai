1. Project Overview and Architecture

This documentation outlines the architecture for a modern, secure web application built entirely in Rust, leveraging the language's memory safety guarantees and performance characteristics for both frontend and backend development. The system implements a microservices architecture pattern with strong security foundations and modern UX/UI principles.

1.1 Core Architecture Principles

Memory Safety: Rust's ownership system prevents common vulnerabilities Zero-Cost Abstractions: High-level features without runtime overhead Concurrent Safety: Built-in protection against data races
Modular Design: Clean separation of concerns with well-defined interfaces
Security-First: Defense in depth with multiple security layers

5.1 Theme System Architecture

The theme system is built using CSS custom properties (variables) and Rust state management, providing seamless switching between light and dark modes with user preference persistence.

// Theme management hook use yew::prelude::*;
use web_sys::window;

#[derive(Clone, PartialEq)] pub enum Theme {
Light, Dark,
}

impl Theme {
pub fn toggle(&self) -> Self { match self {
Theme::Light => Theme::Dark, Theme::Dark => Theme::Light,
}
}

pub fn css_class(&self) -> &'static str { match self {
Theme::Light => "theme-light",
Theme::Dark => "theme-dark",
}
}
}

#[hook]
pub fn use_theme() -> (Theme, Callback) { let theme = use_state(|| {
// Load theme from localStorage or detect system preference if let Some(storage) =
window().unwrap().local_storage().unwrap() {
if let Ok(Some(saved_theme)) = storage.get_item("theme") {

match saved_theme.as_str() { "dark" => return Theme::Dark,
_ => return Theme::Light,
}
}
}

// Detect system preference if window().unwrap()
.match_media("(prefers-color-scheme: dark)")
.unwrap()
.unwrap()
.matches() { Theme::Dark
} else {
Theme::Light
}
});

let toggle_theme = {
let theme = theme.clone(); Callback::from(move |_| {
let new_theme = (*theme).toggle();

// Save to localStorage if let Some(storage) =
window().unwrap().local_storage().unwrap() {
let _ = storage.set_item("theme", match new_theme {
Theme::Dark => "dark", Theme::Light => "light",
}
);
}

theme.set(new_theme);
})
};

((*theme).clone(), toggle_theme)
}
